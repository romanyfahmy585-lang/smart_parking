      now.getMinutes().toString().padStart(2,'0') + ':' +
      now.getSeconds().toString().padStart(2,'0');

// ═══════════════════════════════════════
// Build JSON
// ═══════════════════════════════════════
String buildStatusJSON() {
  String json = "{";
  for (int i = 1; i <= NUM_SLOTS; i++) {
    if (i > 1) json += ",";
    json += "\"" + String(i) + "\":\"";
    json += (slotStatus[i].length() ? slotStatus[i] : "FREE");
    json += "\"";
  }
  json += "}";
  return json;
}

// ═══════════════════════════════════════
// Handlers
// ═══════════════════════════════════════
void handlePortal() {
  webServer.send_P(200, "text/html", PORTAL_HTML);
}

void handleStatus() {
  webServer.sendHeader("Access-Control-Allow-Origin", "*");
  webServer.send(200, "application/json", buildStatusJSON());
}

void handleUpdate() {
  if (webServer.hasArg("plain")) {
    String body = webServer.arg("plain");
    int si = body.indexOf("\"slot\":");
    int ss = body.indexOf("\"status\":\"");
    if (si != -1 && ss != -1) {
      int slot  = body.substring(si + 7).toInt();
      int start = ss + 10;
      int end   = body.indexOf("\"", start);
      String st = body.substring(start, end);
      st.toUpperCase();
      if (slot >= 1 && slot <= NUM_SLOTS) {
        slotStatus[slot] = st;
        Serial.println("Slot " + String(slot) + " = " + st);
      }
    }
  }
  webServer.send(200, "application/json", "{\"ok\":true}");
}

// ═══════════════════════════════════════
// Setup
// ═══════════════════════════════════════
void setup() {

  Serial.begin(115200);

  // UART2 — RX2=GPIO16, TX2=GPIO17
  ArduinoSerial.begin(9600, SERIAL_8N1, 16, 17);

  // ── Access Point ──
  WiFi.mode(WIFI_AP);
  WiFi.softAPConfig(AP_IP, AP_IP, AP_SUBNET);
  WiFi.softAP(AP_SSID, AP_PASS);

  Serial.println("\nAP Started");
  Serial.print("SSID : "); Serial.println(AP_SSID);
  Serial.print("IP   : "); Serial.println(WiFi.softAPIP());

  // ── DNS Catchall ──
  dnsServer.start(53, "*", AP_IP);

  // ── Routes ──
  webServer.on("/",                         HTTP_GET,  handlePortal);
  webServer.on("/api/status",               HTTP_GET,  handleStatus);
  webServer.on("/update",                   HTTP_POST, handleUpdate);
  webServer.on("/generate_204",             HTTP_GET,  handlePortal);
  webServer.on("/gen_204",                  HTTP_GET,  handlePortal);
  webServer.on("/connecttest.txt",          HTTP_GET,  handlePortal);
  webServer.on("/redirect",                 HTTP_GET,  handlePortal);
  webServer.on("/hotspot-detect.html",      HTTP_GET,  handlePortal);
  webServer.on("/library/test/success.html",HTTP_GET,  handlePortal);
  webServer.on("/success.txt",              HTTP_GET,  handlePortal);
  webServer.onNotFound(handlePortal);

  webServer.begin();
  Serial.println("Web Server Running");
}

// ═══════════════════════════════════════
// Loop
// ═══════════════════════════════════════
void loop() {

  dnsServer.processNextRequest();
  webServer.handleClient();

  // ── Arduino UART ──
  if (ArduinoSerial.available()) {
    String msg = ArduinoSerial.readStringUntil('\n');
    msg.trim();
    Serial.println("Received: " + msg);

    int colonIndex = msg.indexOf(':');
    if (colonIndex != -1) {
      int    slot   = msg.substring(0, colonIndex).toInt();
      String status = msg.substring(colonIndex + 1);
      status.trim();
      status.toUpperCase();

      if (slot >= 1 && slot <= NUM_SLOTS) {
        slotStatus[slot] = status;
        Serial.println("Slot " + String(slot) + " = " + status);
      }
    }
  }
}
